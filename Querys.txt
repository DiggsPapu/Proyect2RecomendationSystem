	Filtering algorithms

Relationship between sectors and users
match(a:user),(b:business) 
where a.TypeBusiness1=b.sector OR a.TypeBusiness2=b.sector OR a.TypeBusiness3=b.sector 
Create(a)-[r:InterestedInTheSectorOf{name: a.name +'<->' + b.name}]->(b)
return type (r), r.name

Relationship between business sector and business
match(a:business),(b:business) 
where a.sector=b.sector AND a.name<>b.name AND a.PrecioBeneficio<>b.PrecioBeneficio 
Create(a)-[r:InTheSameSectorAs{name: a.name +'<->' + b.name}]->(b)
return type (r), r.name

Relationship between if wants to be a big business or not
match(a:user),(b:business) 
where a.size='muy grande' AND b.capital>=50 OR a.size='grande' AND b.capital<50 AND b.capital>=30 OR a.size='mediana' AND b.capital<30 AND b.capital>=15 OR a.size='little' AND b.capital<15 AND b.capital>=3  OR a.size='underdog' AND b.capital<3
Create(a)-[r:TheSizeTheyWant{name: a.name +'<->' + b.name}]->(b)
return type (r), r.name

Relationship of risk
match(a:user),(b:business) 
where a.Risk= 5 AND b.var< -1 OR a.Risk=4 AND b.var<0 AND b.var>=-1 OR a.Risk = 3 AND b.var<2 AND b.var >= 0 OR a.Risk = 2 AND b.var<5 AND b.var>=2  OR a.Risk = 1 AND b.var>5
Create(a)-[r:TheRiskTheyWant{name: a.name +'<->' + b.name}]->(b)
return type (r), r.name

Relationship of rentability
match(a:user),(b:business) 
where a.rentability= 5 AND b.rentabilidad>30 OR a.rentability=4 AND b.rentabilidad<=30 AND b.rentabilidad>15 OR a.rentability = 3 AND b.rentabilidad<=15 AND b.rentabilidad > 10 OR a.rentability = 2 AND b.rentabilidad<=10 AND b.rentabilidad>=5  OR a.rentability = 1 AND b.rentabilidad<5 OR b.rentabilidad = null

return type (r), r.name
